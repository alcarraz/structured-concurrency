\section{Scoped Values}

\begin{frame}
    \frametitle{Scoped Values = Contexto Sin Parameter Drilling}

    \begin{exampleblock}{El Problema}
        \textbf{¿Cómo pasar contexto (audit info, request ID, user context) a través de validaciones paralelas?}
        \begin{itemize}
            \item \textbf{Parameter drilling}: Pasar en cada firma de método (verbose, invasivo)
            \item \textbf{ThreadLocal}: Problemático con hilos virtuales masivos
        \end{itemize}
    \end{exampleblock}

    \begin{block}{La Solución: Scoped Values}
        \begin{itemize}
            \item \textbf{Propagación automática}: \texttt{ScopedValue.where(KEY, value).run(...)}
            \item \textbf{Acceso directo}: \texttt{KEY.get()} desde cualquier método
            \item \textbf{Thread-safe}: Funciona perfectamente con structured concurrency
            \item \textbf{Casos de uso}: Audit trails, request IDs, user context, tracing
        \end{itemize}
    \end{block}
